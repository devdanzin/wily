05 01 --- --- --- --- --- --- ------- ------- ------- |     def get_cyclomatic_color(self, val: int) -> str:

05 01 --- --- --- --- --- --- ------- ------- ------- |         """Calculate RGB values for a scale from green to red through yellow."""

05 01 --- --- --- --- --- --- ------- ------- ------- |         # This works well for values where red means 50 or above, like Cyclomatic

05 01 --- --- --- --- --- --- ------- ------- ------- |         # Complexity. For other ranges, different values should be used.

05 01 --- --- --- --- --- --- ------- ------- ------- |         red = max(0, min(255, round(0.04 * 255 * (val - 1))))

05 01 --- --- --- --- --- --- ------- ------- ------- |         green = max(0, min(255, round(0.04 * 255 * (50 - val + 1))))

05 01 --- --- --- --- --- --- ------- ------- ------- |         blue = 0

05 01 --- --- --- --- --- --- ------- ------- ------- |         return f"rgba{(red, green, blue, 0.75)}"
